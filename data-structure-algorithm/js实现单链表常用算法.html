# js实现单链表常用算法

// 1.定义单链表节点Node
function Node(val) {
    this.val = val
    this.next = null
}

// 考虑到有些小伙伴对this不太敏感，这里说一个技巧（代入法）：
// 上面的Node构造函数，比如实例化了一个新节点node,那你就把node带入this，想象成node.val和node.next
// 下面的List构造函数也一样的
// 最后是成员函数的this，哪个链表（比如myList）调用这个成员函数，你就把myList带入this


// 2.定义链表及其常用函数（函数都没有考虑边界值，默认链表非空，输入值合法）
function List() {
    // 链表的头节点
    this.head = new Node('head')
    // 新建链表
    this.newList = newList
    // 增
    this.inserte = inserte
    // 删
    this.deletee = deletee
    // 查
    this.finde = finde
    // 查最前一个
    this.findPre = findPre
    // 打印
    this.printe = printe
    // 倒数第K个
    this.findK = findK
    // 判断是否有环
    this.huane = huane
    // 查找最后那个
    this.findLast = findLast
    // 尾插法实现反转
    this.reverse1 = reverse1
    // 3指针实现反转
    this.reverse2 = reverse2
    // 去除重复元素
    this.removeDuplicateNode = removeDuplicateNode
    
    // 最后还有一个合并两个有序数组
}

// 根据输入的数组建立新链表（尾插法）
function newList(arr) {
    let p = this.head
    arr.forEach(item => {
        let newNode = new Node(item)
        p.next = newNode
        p = newNode
    })
}

// 插入，在值为val1的节点后插入
function inserte(val1, val2) {
    // 先找到值为val1的节点
    let p = this.finde(val1)
    if (p === -1) {
        alert(p)
    }
    let newNode = new Node(val2)
    newNode.next = p.next
    p.next = newNode
}

// 删除
function deletee(val) {
    let p = this.finde(val)
    if (p === -1) {
        alert(p)
    }
    let pre = this.findPre(p)
    pre.next = p.next
}

// 查找
function finde(val) {
    let p = this.head
    while (p.next) {
        p = p.next
        if (p.val === val) return p
    }
    return -1
}

// 查找前一个节点
function findPre(node) {
    let p = this.head
    while (p.next) {
        if (p.next === node) {
            return p
        }
        p = p.next
    }
}

// 打印链表
function printe() {
    let p = this.head
    while (p.next) {
        console.log(p.next.val)
        p = p.next
    }
}

// 输出倒数第k个值【双指针（‘快慢’指针）】
function findK(k) {
    let fast = this.head
    let slow = this.head
    while (k--) {
        fast = fast.next
    }
    while (fast) {
        fast = fast.next
        slow = slow.next
    }
    return slow.val
}

// 判断是否有环【快慢指针】
function huane() {
    let fast = this.head
    let slow = this.head

    while (fast.next) {
        fast = fast.next.next
        slow = slow.next
        if (slow === fast) {
            return 1
        }
    }
    return -1
}

// 最后的节点
function findLast() {
    let p = this.head
    while (p.next) {
        p = p.next
    }
    return p
}

// 1.头插法实现链表反转（头插法想象成两个节点【前一个节点永远是head】中间插一个节点，就很好理解了）
// 再说一句，有些文章挺误导人的，链表哪来的啥原地不原地的。。。节点都可以复用的嘛
function reverse1() {
    let _head = this.head
    let p = _head.next
    _head.next = null
    while (p) {
        let q = p.next
        p.next = _head.next
        _head.next = p
        p = q
    }
}

// 2.三指针法实现链表反转，每次反转两个节点
function reverse2() {
    // 头不能一起跟着反转
    let mid = this.head.next
    let pre = null
    let _next = null

    while (mid.next) {
        // 先用next保存mid.next
        _next = mid.next
        // 反转两个节点
        mid.next = pre
        // 往后走，继续反转下一组节点
        pre = mid
        // 往后走，继续反转下一组节点
        mid = _next
    }
    // 最后一次反转
    mid.next = pre
    // 把头拉过来
    this.head.next = mid
}

// 排序链表去重
function removeDuplicateNode() {
    let p = this.head.next
    while (p) {
        // 要判断p.next是否为空，是就到尾了
        while (p.next && p.val === p.next.val) {
            p.next = p.next.next
        }
        p = p.next
    }
}

// 合并两个有序链表
function merge(list1, list2) {
    let p1 = list1.head.next
    let p2 = list2.head.next
    let p = list1.head
    list2.head.next = null

    while (p1 && p2) {
        if (p1.val <= p2.val) {
            p.next = p1
            p = p1
            p1 = p1.next
        } else {
            p.next = p2
            p = p2
            p2 = p2.next
        }
    }
    if (p1) {
        p.next = p1
    } else p.next = p2

    return list1
}
